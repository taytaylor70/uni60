Quick Start

Running in three lines. Inspectable by design.

Every HEXAGEN-60 ID carries its own metadata — type, timestamp, sequence, and checksum — readable without a database lookup. No external state required.

quickstart.py
import uni60 from uni60 import generate, inspect, TypePrefix
# Generate typed, production-grade IDs user_id = generate(TypePrefix.USER) event_id = generate(TypePrefix.EVENT) asset_id = generate(TypePrefix.ASSET)
# Decode any ID — no database required info = inspect(user_id) print(info.type_name) # 'USER' print(info.era) # 730 (days since epoch) print(info.period) # 43200 (seconds into today) print(info.checksum_valid) # True
# Raw Base-60 encoding uni60.encode(3600) # '100' uni60.decode('100') # 3600
